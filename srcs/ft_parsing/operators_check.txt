echo Hello | > ||

echo Hello | > |||||

echo Hello | > <
bash: syntax error near unexpected token `<'

echo Hello | > <<<<
bash: syntax error near unexpected token `<<<'

echo Hello | > >>>
bash: syntax error near unexpected token `>>'

echo Hello | > &
bash: syntax error near unexpected token `&'

echo Hello | > &&&
bash: syntax error near unexpected token `&&'
